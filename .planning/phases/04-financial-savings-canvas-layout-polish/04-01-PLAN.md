**RECOMMENDED APPROACH — hourly bucketed TOU savings:**
```sql
SELECT
  SUM(
    CASE
      WHEN EXTRACT(DOW FROM bucket AT TIME ZONE 'Asia/Bangkok') IN (0, 6) THEN hourly_kwh * 2.6369
      WHEN EXTRACT(HOUR FROM bucket AT TIME ZONE 'Asia/Bangkok') >= 9
       AND EXTRACT(HOUR FROM bucket AT TIME ZONE 'Asia/Bangkok') < 22 THEN hourly_kwh * 5.7982
      ELSE hourly_kwh * 2.6369
    END
  ) AS savings_thb
FROM (
  SELECT
    DATE_BIN(INTERVAL '1 hour', time, '1970-01-01T00:00:00Z') AS bucket,
    MAX(COALESCE(today_production_1_sensor, 0) + COALESCE(today_production_2_sensor, 0) + COALESCE(today_production_3_sensor, 0) + COALESCE(today_production_4_sensor, 0))

  FROM "East Microinverter"
  WHERE time >= now() - INTERVAL '24 hours'
  GROUP BY 1
) WHERE hourly_kwh > 0
```

  ```sql
  SELECT SUM(
    CASE
      WHEN EXTRACT(DOW FROM bucket AT TIME ZONE 'Asia/Bangkok') IN (0, 6) THEN hourly_kwh * 2.6369
      WHEN EXTRACT(HOUR FROM bucket AT TIME ZONE 'Asia/Bangkok') >= 9

      ELSE hourly_kwh * 2.6369

    SELECT
      DATE_BIN(INTERVAL '1 hour', time, '1970-01-01T00:00:00Z') AS bucket,
      AVG(COALESCE(pv1_power_sensor, 0) + COALESCE(pv2_power_sensor, 0) + COALESCE(pv3_power_sensor, 0) + COALESCE(pv4_power_sensor, 0)) / 1000.0 AS hourly_kwh
    FROM "East Microinverter"
    WHERE time >= date_trunc('month', now())
    GROUP BY 1
  ) WHERE hourly_kwh > 0
  ```
  
  This uses AVG(total PV power in W) per hour / 1000 to get kWh per hour. The division gives kWh directly since it's avg W over 1 hour.
